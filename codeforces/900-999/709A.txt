#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, b, d;
    cin >> n >> b >> d;

    int waste = 0;
    int ans = 0;

    for (int i = 0; i < n; i++) {
        int a;
        cin >> a;

        if (a > b) continue;     
        waste += a;

        if (waste > d) {
            ans++;
            waste = 0;         
        }
    }

    cout << ans << "\n";
    return 0;
}
